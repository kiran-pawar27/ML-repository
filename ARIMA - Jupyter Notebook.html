      def adf_test(series):
          result = adfuller(series)
          print("ADF stats : {}".format(result[0]))
          print("P - Value: {}" .format(result[1]))
          if(result[1]<=0.05):
              print(" Reject Null hypothesis. Data is statonary")
          else:
              print("Accpet Null hypothesis. Data is not stationary")
